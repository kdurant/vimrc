while(SciaRegs.SCIFFRX.bit.RXFFST != 0) //receive fifo number, max is 4.
    {
        ch = (SciaRegs.SCIRXBUF.all & 0xff) ;
        gCyclicBufWrPtr ++ ;
        if(gCyclicBufWrPtr >= CYCLIC_BUF_LEN)
        {
            gCyclicBufWrPtr = 0 ;
        }
        gCyclicBuf[gCyclicBufWrPtr] = ch ;
    }

void CyclicBufProcess(void)
{
    if(gOneFrameReceived == 0 )
    {
        if(gHeadReceivedFlag == 0)
        {
            gCyclicBufRdPtr ++ ;
            if(gCyclicBuf[gCyclicBufRdPtr] == UART_HEAD) //0xFE
            {
                gSciRecNum = 0 ;
                gHeadReceivedFlag = 1 ;
                gSciRecBuf[gSciRecNum] = gCyclicBuf[gCyclicBufRdPtr] ;
                gSciRecNum ++ ;
                if (gCyclicBufRdPtr >= CYCLIC_BUF_LEN)
                {
                    gCyclicBufRdPtr = 0 ;
                }
            }
        }
        else
        {
            gCyclicBufRdPtr ++ ;
            if (gCyclicBufRdPtr >= CYCLIC_BUF_LEN)
            {
                gCyclicBufRdPtr = 0 ;
            }
            gSciRecBuf[gSciRecNum++] = gCyclicBuf[gCyclicBufRdPtr] ;
            if(gSciRecNum >= UART_RECBUF_LEN)
            {
                gSciRecNum = 0 ;
                gHeadReceivedFlag = 0 ;
                gTotalToReceive = 0 ;
            }
            else
            {
                if(gSciRecNum == 2)
                {
                    gTotalToReceive = gCyclicBuf[gCyclicBufRdPtr] ; //data frame length
                }
                else if(gTotalToReceive == gSciRecNum) //The one data frame received is completed
                {
                    gOneFrameReceived = 1 ;
                    Sci_OneFramProcess() ;
                }
            }
        }
    }
}
