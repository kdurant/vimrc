unsigned int factorial(int i)
{
    int ans ;
    if(i > 0)
    {
        printf("i = %d\n", i ) ;
        ans = i * factorial(i-1) ;
    }
    else
    {
        ans = 1 ;
        //printf("string\n") ;
    }
    return ans ;
}
