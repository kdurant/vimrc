    task HarwareReset ;
    input   [3:0]   HoldLowTime ;
    begin
        RSTn = 1'b1;
        wait (Clk !== 1'bx);
        @ (negedge Clk);
        RSTn <= 1'b0;
        repeat(HoldLowTime)
            @ (negedge Clk);
        RSTn <= 1'b1;
    end
endtask
